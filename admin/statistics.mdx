## Overview

The statistics module provides comprehensive analytics for tracking user behavior, fixer performance, and search patterns across the Servineo platform.

<Note>
  Statistics are fetched from the backend API and support multiple time periods: weekly (semanal), monthly (mensual), and yearly (anual).
</Note>

## Statistics API Integration

Fetch statistics using Redux RTK Query:

```typescript
import { useGetStatisticsQuery } from '@/app/redux/services/dashboardApi';

function StatisticsPage() {
  const [selectedPeriod, setSelectedPeriod] = useState<'semanal' | 'mensual' | 'anual'>('mensual');
  const { data, error, isLoading, refetch } = useGetStatisticsQuery(selectedPeriod);
  
  useEffect(() => {
    refetch();
  }, [selectedPeriod, refetch]);
  
  if (isLoading) return <div>Loading statistics...</div>;
  if (error) return <div>Error loading data</div>;
  
  return (
    <div>
      <h2>Total Searches: {data.count}</h2>
      {/* Render charts */}
    </div>
  );
}
```

## Data Structure

Statistics API returns the following structure:

```typescript
interface StatisticsData {
  count: number; // Total searches in period
  filter_statistics: {
    fixer_name: Record<string, number>; // Searches by fixer
    city: Record<string, number>;        // Searches by city
    job_type: Record<string, number>;    // Searches by job type
  };
  user_type_statistics: Record<string, number>; // Distribution by user type
}
```

## Chart Components

### Pie/Donut Chart - User Distribution

Visualize user type distribution:

```tsx
import { ChartPieDonut } from '@/Components/Statistics-panel/Chart-pie-donut';

function UserDistributionChart({ data }: { data: StatisticsData }) {
  const pieData = transformFilterData(data.user_type_statistics);
  
  return (
    <ChartPieDonut
      data={pieData}
      title="User Distribution"
      description="Breakdown by user type"
    />
  );
}

// Transform backend data to chart format
const transformFilterData = (filterData: Record<string, number>) => {
  return Object.entries(filterData).map(([name, value]) => ({
    name,
    value,
    fill: getColorForCategory(name)
  }));
};
```

<Note>
  The pie chart uses Recharts library with customizable colors and displays percentages automatically.
</Note>

## Period Selection

<Steps>
  <Step title="Choose Period">
    Select from weekly, monthly, or yearly views:
    
    ```tsx
    <select
      value={selectedPeriod}
      onChange={(e) => setSelectedPeriod(e.target.value)}
    >
      <option value="semanal">Weekly</option>
      <option value="mensual">Monthly</option>
      <option value="anual">Yearly</option>
    </select>
    ```
  </Step>
  
  <Step title="Auto-Refresh Data">
    Statistics automatically refetch when period changes:
    
    ```typescript
    useEffect(() => {
      refetch();
    }, [selectedPeriod, refetch]);
    ```
  </Step>
  
  <Step title="Display Results">
    Show total count and period label:
    
    ```tsx
    <h2>Period: {selectedPeriod === 'semanal' ? 'Weekly' : 'Monthly'}</h2>
    <p>Total Searches: {data.count}</p>
    ```
  </Step>
</Steps>

## Chart Customization

Customize bar charts with various options:


## Real-World Example

Complete statistics page implementation:

```tsx
'use client';
import { useState, useEffect } from 'react';
import { useGetStatisticsQuery } from '@/app/redux/services/dashboardApi';
import { ChartPieDonut } from '@/Components/Statistics-panel/Chart-pie-donut';
import { ChartBarLabel } from '@/Components/Statistics-panel/Chart-bar-Label';

export default function StatisticsPage() {
  const [period, setPeriod] = useState<'semanal' | 'mensual' | 'anual'>('mensual');
  const { data, isLoading, error, refetch } = useGetStatisticsQuery(period);
  
  useEffect(() => {
    refetch();
  }, [period, refetch]);
  
  if (isLoading) return <LoadingSpinner />;
  if (error) return <ErrorMessage />;
  
  const userTypeData = transformData(data.user_type_statistics);
  const fixerData = transformData(data.filter_statistics.fixer_name);
  const cityData = transformData(data.filter_statistics.city);
  const jobTypeData = transformData(data.filter_statistics.job_type);
  
  return (
    <main className="p-8">
      <PeriodSelector value={period} onChange={setPeriod} />
      
      <h2>Total Searches: {data.count}</h2>
      
      {userTypeData.length > 0 && (
        <ChartPieDonut 
          data={userTypeData}
          title="User Distribution"
        />
      )}
      
      {fixerData.length > 0 && (
        <ChartBarLabel
          data={fixerData}
          title="Top Fixers"
          color="#2563eb"
        />
      )}
      
      {cityData.length > 0 && (
        <ChartBarLabel
          data={cityData}
          title="Cities"
          color="#10b981"
        />
      )}
      
      {jobTypeData.length > 0 && (
        <ChartBarLabel
          data={jobTypeData}
          title="Job Types"
          color="#f59e0b"
        />
      )}
    </main>
  );
}
```
