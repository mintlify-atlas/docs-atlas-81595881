## Overview

The promotions system allows administrators and fixers to create, manage, and delete promotional offers for specific job postings. Each promotion is linked to a job offer and a fixer.

<Note>
  Promotions are tied to specific job offers and fixers. They help fixers stand out and attract more clients.
</Note>

## Promotion Structure

```typescript
interface Promotion {
  _id: string;
  title: string;
  description: string;
  offerId: string;  // Job offer ID
  fixerId: string;  // Fixer who created the promotion
  createdAt?: string;
  updatedAt?: string;
}

interface CreatePromotion {
  title: string;
  description: string;
  offerId: string;
  fixerId: string;
  price?: string;
}
```

## Fetching Promotions by Job Offer

Retrieve all promotions for a specific job:

```typescript
import { getPromotionsByOfferId } from '@/services/promotions';

async function loadPromotions(jobId: string) {
  try {
    const promotions = await getPromotionsByOfferId(jobId);
    console.log(`Found ${promotions.length} promotions`);
    return promotions;
  } catch (error) {
    console.error('Failed to load promotions:', error);
    return [];
  }
}
```

<Warning>
  The API returns an empty array `[]` if no promotions exist or if there's an error. Always check for errors in production.
</Warning>

## Updating Promotions

Edit existing promotion details:

```typescript
import { updatePromotion } from '@/services/promotions';

function EditPromotion({ promotion }: { promotion: Promotion }) {
  const [title, setTitle] = useState(promotion.title);
  const [description, setDescription] = useState(promotion.description);
  
  const handleSave = async () => {
    try {
      const updated = await updatePromotion(promotion._id, {
        title,
        description
      });
      
      if (updated) {
        console.log('Promotion updated successfully');
        // Refresh data
        await loadPromotions(promotion.offerId);
      } else {
        alert('Failed to update promotion');
      }
    } catch (error) {
      console.error('Update error:', error);
      alert('Error updating promotion');
    }
  };
  
  return (
    <div>
      <input value={title} onChange={(e) => setTitle(e.target.value)} />
      <textarea value={description} onChange={(e) => setDescription(e.target.value)} />
      <button onClick={handleSave}>Save Changes</button>
    </div>
  );
}
```

<Note>
  Only the `title` and `description` fields can be updated. The `offerId` and `fixerId` are immutable after creation.
</Note>

## Deleting Promotions

Remove promotions individually or in batch:

### Single Deletion

```typescript
import { deletePromotion } from '@/services/promotions';

async function handleDelete(promotionId: string) {
  const confirmed = confirm('Are you sure you want to delete this promotion?');
  
  if (!confirmed) return;
  
  const success = await deletePromotion(promotionId);
  
  if (success) {
    console.log('Promotion deleted');
    // Remove from state
    setPromotions((prev) => prev.filter((p) => p._id !== promotionId));
  } else {
    alert('Failed to delete promotion');
  }
}
```

### Batch Deletion

```typescript
async function handleBatchDelete(selectedIds: string[]) {
  if (selectedIds.length === 0) {
    alert('No promotions selected');
    return;
  }
  
  try {
    const deletePromises = selectedIds.map((id) => deletePromotion(id));
    const results = await Promise.all(deletePromises);
    
    const successCount = results.filter(Boolean).length;
    console.log(`Deleted ${successCount} of ${selectedIds.length} promotions`);
    
    // Reload promotions
    await loadPromotions(jobId);
    setSelectedIds([]);
  } catch (error) {
    console.error('Batch delete error:', error);
    alert('Error deleting some promotions');
  }
}
```

## Complete Promotion Management Page

Full implementation example from the source code:

```tsx
'use client';
import { useEffect, useState } from 'react';
import { useParams, useRouter } from 'next/navigation';
import { getPromotionsByOfferId, deletePromotion, updatePromotion } from '@/services/promotions';
import { getJobInfo } from '@/services/job-info';
import { Promotion } from '@/types/promotion';
import CreatePromoModal from '@/app/components/fixers/CreatePromoModal';

export default function JobPromotionsPage() {
  const params = useParams();
  const router = useRouter();
  const jobId = params.jobId as string;
  
  const [promotions, setPromotions] = useState<Promotion[]>([]);
  const [jobTitle, setJobTitle] = useState('Job Details');
  const [loading, setLoading] = useState(true);
  const [selectedIds, setSelectedIds] = useState<string[]>([]);
  
  // Load data
  const loadData = async () => {
    setLoading(true);
    try {
      const jobData = await getJobInfo(jobId);
      if (jobData?.title) setJobTitle(jobData.title);
      
      const promosData = await getPromotionsByOfferId(jobId);
      setPromotions(Array.isArray(promosData) ? promosData : []);
    } catch (error) {
      console.error('Error loading data:', error);
    } finally {
      setLoading(false);
    }
  };
  
  useEffect(() => {
    if (jobId) loadData();
  }, [jobId]);
  
  // Toggle selection
  const toggleSelection = (id: string) => {
    setSelectedIds((prev) =>
      prev.includes(id) 
        ? prev.filter((item) => item !== id) 
        : [...prev, id]
    );
  };
  
  // Delete selected
  const handleDeleteConfirm = async () => {
    try {
      const deletePromises = selectedIds.map((id) => deletePromotion(id));
      await Promise.all(deletePromises);
      
      await loadData();
      setSelectedIds([]);
    } catch (error) {
      alert('Error deleting promotions');
      console.error('Delete error:', error);
    }
  };
  
  return (
    <div className="min-h-screen p-8">
      <h1>Promotions - {loading ? '...' : jobTitle}</h1>
      
      {loading ? (
        <LoadingSpinner />
      ) : promotions.length === 0 ? (
        <p>No promotions found for this job.</p>
      ) : (
        <div>
          {promotions.map((promo) => (
            <div key={promo._id}>
              <input
                type="checkbox"
                checked={selectedIds.includes(promo._id)}
                onChange={() => toggleSelection(promo._id)}
              />
              <h3>{promo.title}</h3>
              <p>{promo.description}</p>
            </div>
          ))}
        </div>
      )}
      
      <button
        onClick={handleDeleteConfirm}
        disabled={selectedIds.length === 0}
      >
        Delete Selected ({selectedIds.length})
      </button>
    </div>
  );
}
```

## Modal Components

Use the `CreatePromoModal` component for creating/editing:

```tsx
import CreatePromoModal from '@/app/components/fixers/CreatePromoModal';

function PromotionEditor() {
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [editingPromo, setEditingPromo] = useState<Promotion | null>(null);
  
  const handleSave = async (promotion: CreatePromotion) => {
    if (editingPromo) {
      // Update existing
      await updatePromotion(editingPromo._id, {
        title: promotion.title,
        description: promotion.description
      });
    } else {
      // Create new
      await createPromotion(promotion);

    setEditingPromo(null);
    await loadData();

  return (
    <>
      <button onClick={() => setIsModalOpen(true)}>New Promotion</button>

        initialTitle={editingPromo?.title}
        initialDescription={editingPromo?.description}

    </>
  );
}
```

## Error Handling

Handle API errors gracefully:

```typescript
try {
  const result = await createPromotion(promoData);
  
  if (!result) {
    // API returned null (error occurred)
    throw new Error('Failed to create promotion');
  }
  
  // Success
  console.log('Promotion created:', result._id);
} catch (error) {
  console.error('Error:', error);
  
  // Show user-friendly message
  if (error instanceof Error) {
    alert(`Error: ${error.message}`);
  } else {
    alert('An unexpected error occurred');
  }
}
```
