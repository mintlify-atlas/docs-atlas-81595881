## Overview

Servineo implements comprehensive internationalization (i18n) using `next-intl`, supporting both English and Spanish languages. The system provides seamless language switching with route-based locales and type-safe translation keys.

## Configuration

The routing configuration defines supported locales and the default language:

```typescript routing.ts
import { defineRouting } from 'next-intl/routing';

export const routing = defineRouting({
  // List of all supported locales
  locales: ['en', 'es'],
  
  // Default locale when no locale matches
  defaultLocale: 'en',
});
```

## Using Translations

The `useTranslations` hook provides type-safe access to translations:

<CodeGroup>
```tsx Basic Usage
'use client';

import { useTranslations } from 'next-intl';

export function SearchComponent() {
  const t = useTranslations('search');

  return (
    <div>
      <input 
        placeholder={t('placeholder')} 
        className='w-full px-4 py-2'
      />
      <button>{t('buttonSearch')}</button>
    </div>
  );
}
```

```tsx Nested Keys
'use client';

import { useTranslations } from 'next-intl';

export function JobOffersMap() {
  const t = useTranslations('jobOffers.map');

  return (
    <div className='map-panel'>
      <h3>{t('mapDetails')}</h3>
      <div>
        <span>{t('yourLocation')}</span>
        <span>{t('jobOffers')}</span>
      </div>
    </div>
  );
}
```

```tsx Multiple Namespaces
'use client';

import { useTranslations } from 'next-intl';

export function AdvancedSearchPage() {
  const tSearch = useTranslations('search');
  const tAdvanced = useTranslations('advancedSearch');

  return (
    <div>
      <h1>{tAdvanced('pageTitle')}</h1>
      <input placeholder={tSearch('placeholder')} />
      <label>
        <input type='checkbox' />
        {tSearch('titleOnly')}
      </label>
    </div>
  );
}
```
</CodeGroup>

## Routing Integration

Next-intl integrates with Next.js routing for locale-based URLs:

```typescript navigation.ts
import { createLocalizedPathnamesNavigation } from 'next-intl/navigation';
import { routing } from './routing';

export const { Link, redirect, usePathname, useRouter } = 
  createLocalizedPathnamesNavigation(routing);
```

### Usage in Components

```tsx
import { Link, useRouter } from '@/i18n/navigation';

export function Navigation() {
  const router = useRouter();

  return (
    <nav>
      {/* Automatic locale prefix */}
      <Link href='/job-offers'>Job Offers</Link>
      <Link href='/advanced-search'>Advanced Search</Link>
      
      <button onClick={() => router.push('/profile')}>
        Profile
      </button>
    </nav>
  );
}
```

## Language Switcher

Implement a language switcher to allow users to change locales:

```tsx LanguageSwitcher.tsx
'use client';

import { useLocale } from 'next-intl';
import { useRouter, usePathname } from '@/i18n/navigation';

export function LanguageSwitcher() {
  const locale = useLocale();
  const router = useRouter();
  const pathname = usePathname();

  const switchLocale = (newLocale: string) => {
    router.replace(pathname, { locale: newLocale });
  };

  return (
    <div className='flex gap-2'>
      <button
        onClick={() => switchLocale('en')}
        className={`px-3 py-1 rounded ${
          locale === 'en' ? 'bg-primary text-white' : 'bg-gray-200'
        }`}
      >
        English
      </button>
      <button
        onClick={() => switchLocale('es')}
        className={`px-3 py-1 rounded ${
          locale === 'es' ? 'bg-primary text-white' : 'bg-gray-200'
        }`}
      >
        Español
      </button>
    </div>
  );
}
```

## Server-Side Configuration

Configure next-intl in your Next.js middleware:

```typescript middleware.ts
import createMiddleware from 'next-intl/middleware';
import { routing } from './i18n/routing';

export default createMiddleware(routing);

export const config = {
  matcher: ['/', '/(en|es)/:path*']
};
```

## Request Configuration

Configure how next-intl retrieves the locale for server components:

```typescript request.ts
import { getRequestConfig } from 'next-intl/server';
import { routing } from './routing';

export default getRequestConfig(async ({ requestLocale }) => {
  let locale = await requestLocale;

  // Validate that the incoming locale parameter is valid
  if (!locale || !routing.locales.includes(locale as any)) {
    locale = routing.defaultLocale;
  }

  return {
    locale,
    messages: (await import(`../messages/${locale}.json`)).default,
  };
});
```

## Dynamic Translations

For dynamic content, use interpolation:

```tsx
// In translation file
{
  "welcome": "Welcome, {name}!",
  "itemsCount": "You have {count} items"
}

// In component
const t = useTranslations('messages');

<p>{t('welcome', { name: userName })}</p>
<p>{t('itemsCount', { count: items.length })}</p>
```

## Rich Text Formatting

Support rich text with next-intl's formatting:

```tsx
const t = useTranslations('messages');

<p>
  {t.rich('agreement', {
    link: (chunks) => <a href='/terms'>{chunks}</a>,
    strong: (chunks) => <strong>{chunks}</strong>
  })}
</p>
```

## Date and Number Formatting

Format dates and numbers according to locale:

```tsx
import { useFormatter } from 'next-intl';

export function DateDisplay({ date }: { date: Date }) {
  const format = useFormatter();

  return (
    <div>
      <p>{format.dateTime(date, { dateStyle: 'long' })}</p>
      <p>{format.number(1234.56, { style: 'currency', currency: 'USD' })}</p>
    </div>
  );
}
```

## Common Patterns

### Search Checkboxes Example

Real implementation from the codebase:

```tsx SearchCheckboxes.tsx
'use client';

import { useTranslations } from 'next-intl';

export function SearchCheckboxes({
  titleOnly,
  setTitleOnly,
  exactWords,
  setExactWords,
}: SearchCheckboxesProps) {
  const t = useTranslations('search');

  return (
    <div className='w-full bg-gray-100 border border-gray-200 rounded-lg p-4 mt-4'>
      <label className='flex items-center gap-3 cursor-pointer hover:bg-gray-50 p-2 rounded'>
        <input
          type='checkbox'
          checked={titleOnly}
          onChange={(e) => setTitleOnly(e.target.checked)}
          className='w-4 h-4 text-blue-600 border-gray-300 rounded'
        />
        <span className='text-sm text-gray-700'>{t('titleOnly')}</span>
      </label>

      <label className='flex items-center gap-3 cursor-pointer hover:bg-gray-50 p-2 rounded'>
        <input
          type='checkbox'
          checked={exactWords}
          onChange={(e) => setExactWords(e.target.checked)}
          className='w-4 h-4 text-blue-600 border-gray-300 rounded'
        />
        <span className='text-sm text-gray-700'>{t('exactWords')}</span>
      </label>
    </div>
  );
}
```
