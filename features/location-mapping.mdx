## Basic Map View

The basic map view allows users to select a location by clicking on the map.

```tsx MapView.tsx
'use client';

import { MapContainer, TileLayer, Marker, useMapEvents } from 'react-leaflet';
import { useState, useEffect } from 'react';
import 'leaflet/dist/leaflet.css';

interface MapViewProps {
  onSelect: (lat: number, lon: number) => void;
  markerPosition?: { lat: number; lon: number } | null;
}

export default function MapView({ onSelect, markerPosition }: MapViewProps) {
  return (
    <MapContainer
      center={[markerPosition?.lat || -17.39, markerPosition?.lon || -66.15]}
      zoom={13}
      style={{ height: 400, width: '100%' }}
    >
      <TileLayer url='https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png' />
      <MarkerHandler onSelect={onSelect} markerPosition={markerPosition} />
    </MapContainer>
  );
}

function MarkerHandler({ onSelect, markerPosition }: MarkerHandlerProps) {
  const [position, setPosition] = useState(markerPosition || null);

  useEffect(() => {
    setPosition(markerPosition || null);
  }, [markerPosition]);

  useMapEvents({
    click(e) {
      setPosition({ lat: e.latlng.lat, lon: e.latlng.lng });
      onSelect(e.latlng.lat, e.latlng.lng);
    },
  });

  return position ? <Marker position={[position.lat, position.lon]} /> : null;
}
```

<Note>
The map uses Bolivia's coordinates as default: latitude -17.39, longitude -66.15 (Cochabamba)
</Note>

### Distance Calculation

The Haversine formula calculates distances between coordinates:

```typescript
const calculateDistance = (lat1: number, lng1: number, lat2: number, lng2: number) => {
  const R = 6371; // Earth's radius in kilometers
  const dLat = ((lat2 - lat1) * Math.PI) / 180;
  const dLng = ((lng2 - lng1) * Math.PI) / 180;
  
  const a =
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos((lat1 * Math.PI) / 180) *
      Math.cos((lat2 * Math.PI) / 180) *
      Math.sin(dLng / 2) *
      Math.sin(dLng / 2);
  
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  return R * c; // Distance in kilometers
};
```

## Hover Interactions

The map features sophisticated hover interactions with debouncing to prevent flickering:

```typescript
const hoverTimeoutRef = useRef<NodeJS.Timeout | null>(null);
const isHoveringCardRef = useRef<boolean>(false);
const currentHoveredOfferIdRef = useRef<string | null>(null);
const pendingHoverRef = useRef<string | null>(null);

markerElement.addEventListener('mouseenter', () => {
  // Clear any pending timeout
  if (hoverTimeoutRef.current) {
    clearTimeout(hoverTimeoutRef.current);
    hoverTimeoutRef.current = null;
  }

  // Skip if already showing this offer
  if (currentHoveredOfferIdRef.current === offer.id) {
    return;
  }

  pendingHoverRef.current = offer.id;

  // Short delay for responsiveness
  hoverTimeoutRef.current = setTimeout(() => {
    if (pendingHoverRef.current === offer.id && !isHoveringCardRef.current) {
      currentHoveredOfferIdRef.current = offer.id;
      setHoveredOffer(offer);
    }
    pendingHoverRef.current = null;
  }, 100);
});

markerElement.addEventListener('mouseleave', () => {
  if (pendingHoverRef.current === offer.id) {
    pendingHoverRef.current = null;
  }

  if (hoverTimeoutRef.current) {
    clearTimeout(hoverTimeoutRef.current);
    hoverTimeoutRef.current = null;
  }

  // Only hide if this is the currently shown marker
  if (currentHoveredOfferIdRef.current === offer.id) {
    hoverTimeoutRef.current = setTimeout(() => {
      if (!isHoveringCardRef.current && currentHoveredOfferIdRef.current === offer.id) {
        currentHoveredOfferIdRef.current = null;
        setHoveredOffer(null);
      }
    }, 150);
  }
});
```

<Info>
The hover system uses refs to track state outside React's render cycle, preventing race conditions and ensuring smooth interactions.
</Info>

## Client-Side Rendering

Leaflet requires client-side rendering. Use this pattern:

```typescript
const [isClient, setIsClient] = useState(false);

useEffect(() => {
  setIsClient(true);
}, []);

if (!isClient) {
  return (
    <div className='w-full h-full rounded-2xl overflow-hidden shadow-lg 
                    bg-gray-200 flex items-center justify-center'>
      <p>Cargando mapa...</p>
    </div>
  );
}
```

## Leaflet Imports

Dynamic imports ensure Leaflet loads only on the client:

```typescript
useEffect(() => {
  if (!isClient || !mapRef.current) return;

  const initializeMap = async () => {
    const L = await import('leaflet');

    const map = L.map(mapRef.current!, {
      zoomControl: false,
    }).setView([userLocation.lat, userLocation.lng], 14);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '© OpenStreetMap contributors',
      maxZoom: 19,
    }).addTo(map);

    // Add markers, circles, etc.
  };

  initializeMap();

  return () => {
    if (mapInstanceRef.current) {
      mapInstanceRef.current.remove();
```
