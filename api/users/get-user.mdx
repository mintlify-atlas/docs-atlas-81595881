## Overview

Retrieve detailed information about a specific user by their ID.

## Endpoint

```
GET /api/user/{id}
```

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the user
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  The user object
  
  <Expandable title="User Object">
    <ResponseField name="_id" type="string">
      User's unique identifier
    </ResponseField>
    
    <ResponseField name="name" type="string">
      User's full name
    </ResponseField>
    
    <ResponseField name="email" type="string">
      User's email address
    </ResponseField>
    
    <ResponseField name="role" type="string">
      User role: `requester`, `fixer`, or `admin`
    </ResponseField>
    
    <ResponseField name="url_photo" type="string">
      URL to user's profile photo
    </ResponseField>
    
    <ResponseField name="telefono" type="string">
      User's phone number
    </ResponseField>
    
    <ResponseField name="ubicacion" type="object">
      User's location information
      
      <Expandable title="Location Details">
        <ResponseField name="lat" type="number">
          Latitude coordinate
        </ResponseField>
        
        <ResponseField name="lng" type="number">
          Longitude coordinate
        </ResponseField>
        
        <ResponseField name="direccion" type="string">
          Street address
        </ResponseField>
        
        <ResponseField name="departamento" type="string">
          Department/state
        </ResponseField>
        
        <ResponseField name="pais" type="string">
          Country
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="workLocation" type="object">
      Fixer's work location (for fixers only)
      
      <Expandable title="Work Location Details">
        <ResponseField name="lat" type="number">
          Latitude coordinate
        </ResponseField>
        
        <ResponseField name="lng" type="number">
          Longitude coordinate
        </ResponseField>
        
        <ResponseField name="direccion" type="string">
          Work address
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="servicios" type="array">
      Array of service IDs the fixer provides
    </ResponseField>
    
    <ResponseField name="vehiculo" type="object">
      Vehicle information
      
      <Expandable title="Vehicle Details">
        <ResponseField name="hasVehiculo" type="boolean">
          Whether the user has a vehicle
        </ResponseField>
        
        <ResponseField name="tipoVehiculo" type="string">
          Type of vehicle
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="metodoPago" type="object">
      Accepted payment methods
      
      <Expandable title="Payment Methods">
        <ResponseField name="hasEfectivo" type="boolean">
          Accepts cash
        </ResponseField>
        
        <ResponseField name="qr" type="boolean">
          Accepts QR code payments
        </ResponseField>
        
        <ResponseField name="tarjetaCredito" type="boolean">
          Accepts credit cards
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="description" type="string">
      User's bio or description
    </ResponseField>
    
    <ResponseField name="createdAt" type="string">
      Account creation timestamp
    </ResponseField>
    
    <ResponseField name="updatedAt" type="string">
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

## Example Request

<CodeGroup>

```bash cURL
curl https://api.servineo.com/api/user/user123 \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN" \
  -H "Content-Type: application/json"
```

```typescript TypeScript
import { userApi } from '@/app/redux/services/userApi';

// Using RTK Query hook
const { data, isLoading, error } = useGetUserByIdQuery('user123');

// Or using lazy query
const [getUser] = useLazyGetUserByIdQuery();
const result = await getUser('user123');
```

```javascript JavaScript
const response = await fetch('https://api.servineo.com/api/user/user123', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer YOUR_ACCESS_TOKEN',
    'Content-Type': 'application/json'
  }
});

const data = await response.json();
```

</CodeGroup>

}
